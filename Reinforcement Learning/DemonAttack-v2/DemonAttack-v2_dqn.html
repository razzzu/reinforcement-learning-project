from keras.models import Model
from keras.layers import Input, Dense, Flatten, Conv3D, MaxPooling3D, concatenate
from keras.utils import plot_model

game_screen = Input((1,)+observation_size.shape)

conv1 = Conv3D(5, kernel_size=(1,25,25),strides=2, activation='relu')(game_screen)
pool1 = MaxPooling3D(pool_size=(1,2,2))(conv1)
conv2 = Conv3D(10, kernel_size = (1,5,5), activation='relu')(pool1)
pool2 = MaxPooling3D(pool_size=(1,2,2))(conv2)
flat_screen = Flatten()(pool2)

move = Dense(100, activation='relu')(flat_screen)
fire = Dense(100, activation='relu')(flat_screen)
merge_move_fire = concatenate([move,fire])

actions=[]
actions.append(Dense(1, activation='linear')(merge_move_fire)) #action_nop
actions.append(Dense(1, activation='linear')(fire)) #action_fire
actions.append(Dense(1, activation='linear')(move)) #action_right
actions.append(Dense(1, activation='linear')(move)) #action_left
actions.append(Dense(1, activation='linear')(merge_move_fire)) # action_fire_right
actions.append(Dense(1, activation='linear')(merge_move_fire)) #action_fire_left 


action_layer=concatenate(actions)
#flat_output=Flatten()(action_layer)

model = Model(inputs=game_screen, outputs=action_layer)
model.summary()
plot_model(model)
